  17 * 

  18 * **hsDatab** provides a JavaScript-based data management and query mechanism.

  19 * Data is managed in a simple in-memory database that holds data in rows of columns. 

  20 * It autodetermines the types of data held in each column, along with the 

  21 * domain range for each column of data. 

  22 * Complex filters can be applied by defining {@link DataFilters `Condition`}s using a simple query object structure. 

  23 * 

  24 * ## Data Types

  25 * supported {@link Data.Data.type data types} include

  26 * - **number**: numeric values

  27 * - **name**: nominal values, represented by arbitrary words

  28 * - **date**: date values

  29 * - **currency**: Currently supported: '$dd[,ddd]'

  30 * - **percent**: 'd%'

  31 * 

  32 * ## Data Class

  33 * The fundamental object in this library is {@link Data.Data `Data`}, 

  34 * a simple row-column based database object, 

  35 * featuring named columns, sorting, mapping and filtering functions.

  36 *

  37 * ## Example

  38 * 

  39 * "script.js">

  40 * const colNames = ['Name', 'Value', 'Start', 'End'];

  41 * const rows = [

  42 *   ['Harry', '100', '3/1/14', '11/20/14'], 

  43 *   ['Mary', '1500', '7/1/14',  '9/30/14'],

  44 *   ['Peter', '400', '5/20/14', '4/30/15'],  

  45 *   ['Jane', '700', '11/13/14', '8/15/15']

  46 * ]

  47 * const data = new hsDatab.Data({colNames:colNames, rows:rows});

  48 * 

  49 * query = {Name:["Peter", "Jane"]};

  50 * 

  51 * const dateConvert = d => new Date(d).toDateString();

  52 *

  53 * m.mount(root, {

  54 *   view:() => m('', [

  55 *       m('h3', 'Given the data set:'),

  56 *       m('pre',

  57 *       m('table#data', [

  58 *           m('tr', colNames.map(n => m('th', n))),

  59 *           ...rows.map(row => m('tr', [

  60 *              m('td', row[0]),

  61 *              m('td', row[1]),

  62 *              m('td', `${row[2].getMonth()+1}/${row[2].getDate()}/${row[2].getFullYear()}`),

  63 *              m('td', `${row[3].getMonth()+1}/${row[3].getDate()}/${row[3].getFullYear()}`)

  64 *          ]))

  65 *       ])),

  66 *       m('h3', 'The column types and domains are'),

  67 *       m('pre', m('table', 

  68 *                  m('tr', m('th', 'Column'),   m('th', 'Type'),   m('th', 'Domain')),

  69 *                  m('tr', m('td', '"Name":'),  m('td', data.colType("Name")),   m('td', data.findDomain("Name").join(', '))),

  70 *                  m('tr', m('td', '"Value":'), m('td', data.colType("Value")),  m('td', data.findDomain("Value").join(' - '))),

  71 *                  m('tr', m('td', '"Start":'), m('td', data.colType("Start")),  m('td', data.findDomain("Start").map(dateConvert).join(' - '))),

  72 *                  m('tr', m('td', '"Stop":'),  m('td', data.colType("End")),    m('td', data.findDomain("End").map(dateConvert).join(' - ')))

  73 *       )),

  74 *       m('h3', 'The query:'),

  75 *       m('code', JSON.stringify(query)),

  76 *       m('h3', 'yields results with "Name"'),

  77 *       m('code', data.filter(query).getColumn('Name').join(', '))

  78 *   ])

  79 * });

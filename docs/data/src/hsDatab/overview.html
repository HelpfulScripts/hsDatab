   1/**

   2 * # hsDatab

   3 * 

   4 * Helpful Scripts framework-independent data management functions. 

   5 * [`[Github page]`](https://github.com/HelpfulScripts/hsDatab)

   6 * [`[Coverage Info]`](./data/src/hsDatab/coverage/)

   7 * ___

   8 * 

   9 * **hsDatab** provides a JavaScript-based data management and query mechanism.

  10 * Data is managed in a simple in-memory database that holds data in rows of columns. 

  11 * It autodetermines the types of data held in each column, along with the 

  12 * domain range for each column of data. 

  13 * Complex filters can be applied by defining {@link DataFilters `Condition`}s using a simple query object structure. 

  14 * 

  15 * ## Data Types

  16 * supported {@link Data.Data.type data types} include

  17 * - **number**: numeric values

  18 * - **name**: nominal values, represented by arbitrary words

  19 * - **date**: date values

  20 * - **currency**: Currently supported: '$dd[,ddd]'

  21 * - **percent**: 'd%'

  22 * 

  23 * ## Data Class

  24 * The fundamental object in this library is {@link Data.Data `Data`}, 

  25 * a simple row-column based database object, 

  26 * featuring named columns, sorting, mapping and filtering functions.

  27 *

  28 * ## Example

  29 * 

  30 * "script.js">

  31 * const colNames = ['Name', 'Value', 'Start', 'End'];

  32 * const rows = [

  33 *   ['Harry', '100', '3/1/14', '11/20/14'], 

  34 *   ['Mary', '1500', '7/1/14',  '9/30/14'],

  35 *   ['Peter', '400', '5/20/14', '4/30/15'],  

  36 *   ['Jane', '700', '11/13/14', '8/15/15']

  37 * ]

  38 * const data = new hsDatab.Data({colNames:colNames, rows:rows});

  39 * 

  40 * query = {Name:["Peter", "Jane"]};

  41 * const result = data.filter(query);

  42 * console.log(data.getData());

  43 * 

  44 * const dateConvert = d => new Date(d).toDateString();

  45 *

  46 * m.mount(root, {

  47 *   view:() => m('', [

  48 *       m('h3', 'Given the data set:'),

  49 *       m('pre',

  50 *       m('table#data', [

  51 *           m('tr', colNames.map(n => m('th', n))),

  52 *           ...rows.map(row => m('tr', [

  53 *              m('td', row[0]),

  54 *              m('td', row[1]),

  55 *              m('td', `${row[2].getMonth()+1}/${row[2].getDate()}/${row[2].getFullYear()}`),

  56 *              m('td', `${row[3].getMonth()+1}/${row[3].getDate()}/${row[3].getFullYear()}`)

  57 *          ]))

  58 *       ])),

  59 *       m('h3', 'The column types and domains are'),

  60 *       m('pre', m('table', 

  61 *                  m('tr', m('th', 'Column'),   m('th', 'Type'),   m('th', 'Domain')),

  62 *                  m('tr', m('td', '"Name":'),  m('td', data.colType("Name")),   m('td', data.findDomain("Name").join(', '))),

  63 *                  m('tr', m('td', '"Value":'), m('td', data.colType("Value")),  m('td', data.findDomain("Value").join(' - '))),

  64 *                  m('tr', m('td', '"Start":'), m('td', data.colType("Start")),  m('td', data.findDomain("Start").map(dateConvert).join(' - '))),

  65 *                  m('tr', m('td', '"Stop":'),  m('td', data.colType("End")),    m('td', data.findDomain("End").map(dateConvert).join(' - ')))

  66 *       )),

  67 *       m('h3', 'The query:'),

  68 *       m('code', '{Name:["Peter", "Jane"]}'),

  69 *       m('h3', 'yields results with "Name"'),

  70 *       m('code', result.getColumn('Name').join(', '))

  71 *   ])

  72 * });

  73 * 

  74 * 'style.css'>

  75 *   $exampleID { height: 600px; }

  76 *   #data th { width:15%; }

  77 *   #data  { font-size: 11pt; }

  78 * 

  79 *

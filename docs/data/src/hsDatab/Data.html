 241        if (domain===undefined) { domain = []; }

 242        if (col === undefined) { // use array index as domain

 243            domain[0] = 0;

 245        } else {

 246            const c = this.colNumber(col);

 247            const type = this.colType(col);

 248            switch(type) {

 249                case Data.type.name: 

 250                    this.data.forEach((r:DataRow) => {

 251                        const nomDom = domain;

 252                        if (nomDom.indexOf(''+r[c]) < 0) { nomDom.push(''+r[c]); }

 253                    });

 254                    break;

 255                case Data.type.date: 

 256                    this.data.forEach((r:DataRow) => {

 257                        let v:Date = this.toDate(r[c]);

 258                        if (domain[0]===undefined) { domain[0] = v; }

 259                        if (domain[1]===undefined) { domain[1] = v; }

 260                        if (v!==undefined && v!==null) {

 261                            if (v

 262                            else if (v>domain[1]) { domain[1] = v; }

 263                        }

 264                    });

 265                    break;

 266                default: 

 267                    this.data.forEach((r:DataRow) => {

 268                        let v:number = parseFloat(r[c].toString());

 269                        if (domain[0]===undefined) { domain[0] = v; }

 270                        if (domain[1]===undefined) { domain[1] = v; }

 271                        if (v!==undefined && v!==null) {

 272                            if (v

 273                            else if (v>domain[1]) { domain[1] = v; }

 274                        }

 275                    });

 276            }

 277        }

 278        return domain;

 302    /**

 303     * @description Sorts the rows of values based on the result of the `sortFn`, 

 304     * which behaves similarly to the Array.sort method.  

 305     * Two modes are supported:

 306     * # Array Mode

 307     * If `col` is omitted, the column arrays are passed as samples into the `sortFn`. 

 308     * This allows for complex sorts, combining conditions across multiple columns.

 309     * ```

 310     * data.sort((row1, row2) => row1[5] - row2[5] );

 311     * ```

 312     * # Column mode

 313     * If `col` is specified, either as index or by column name, the respective column value is passed

 314     * into `sortFn`. This allows filtering for simple conditions.

 315     * **The specified column will be automatically cast prior to sorting**

 316     * ```

 317     * data.sort('ascending', 'Date');

 318     * data.sort('decending', 'Date');

 319     * data.sort(function(val1, val2) { return val1 - val2; }, 'Date');

 320     * ```

 321     * @param col optional; the data column to use for sorting. 

 322     * @param sortFn a function to implement the conditions, 

 323     * follows the same specifications as the function passed to Array.sort(). 

 324     * Some predefined sort function can be invoked by providing a 

 325     * respective string instead of a function. The following functions are defined:

 326        ```

 327        ascending    sort in ascending order.

 328        descending   sort in decending order.

 329        ```

 330     * @return the Data object in order to allow for chaining.

 331     */

 332    public sort(sortFn:string|sortFn, col?:ColumnReference):Data {

 333        let fn = sortFn;

 334        if (!col) {

 335            this.data.sort(fn);

 336        } else {

 337            col = this.colNumber(col);

 338            if (sortFn === 'descending') { fn = (a:any, b:any)  => (b>a)?1:((b

 339            if (sortFn === 'ascending')  { fn = (a:any, b:any)  => (ba)?-1:0); }

 340            this.data.sort((r1:any[], r2:any[]) => fn(r1[col], r2[col])); 

 341        }

 342        return this;

 343    }

 345    /** 

 346    *  Maps one or more columns in each rows of values based 

 347     * on the result of the `mapFn`, which behaves similarly to the Array.map() method.

 348     * Two modes are supported:

 349     * 

 350     * ## Array Mode

 351     * If `col` is omitted, the `mapFn` is passed the column arrays per row as parameter. 

 352     * This allows for complex mapping combining conditions across multiple columns.

 353     * ```

 354     * data.map(function(values){ 

 355     *    values[1] = values[3] * values[5]; 

 356     *    return values; 

 357     * });

 358     * ```

 359     * Be sure to return the `values` array as a result.

 360     * 

 361     * ## Column mode

 362     * If `col` is specified, either as index or by column name, the respective column value is passed

 363     * into `mapFn`, along with the row index and the entire row array. This allows for simple mapping.

 364     * ```

 365     * data.map('Price', function(value, i, values) { 

 366     *    return value * 2; 

 367     * });

 368     * ```

 369     * @param col the data column, or columns, to apply the mapping to. 

 370     * @param mapFn a function to implement the mapping,

 371     * called on each row of the data set in turn as `mapFn(val, i, c, rows)`, where

 372     * - `val`: the column value in the current row

 373     * - `c`: the column index in the current row

 374     * - `i`: the row index 

 375     * - `rows`: the rows being iterated over

 376`    * 

 377     * follows the same specifications as the function passed to Array.map().

 378     * For column mode, some predefined map functions can be invoked by providing a 

 379     * respective string instead of a function. The following functions are defined:

 380     *  

 381     *  

 382     *  

 383     *  

'noop'replace value with itself, performing no operation.
'cumulate'replace value with the cumulative sum of values up to the current element.

 384     *

 385     * @return a new Data object containing the mapping.

 386     */

 387    public map(mapFn:string|mapFn|mapRowFn, col?:ColumnReference):Data {

 388        const noop = (val:any) => val;

 389        const cumulate = () => { 

 390            let sum=0; 

 391            return (val:number, i:number) => { sum += +val; return sum; };

 392        };

 393        function getFn():mapFn|mapRowFn {

 394            let fn; // define fn inside each col loop to ensure initialization

 395            switch (mapFn) {

 396                case 'cumulate': fn = cumulate(); break;

 397                case 'noop':     fn = noop; break;

 398                default:         fn = mapFn;

 399            }

 400            return fn;

 401        }

 402

 403        let fn = getFn(); // define fn inside each col loop to ensure initialization

 404        const c = col? this.colNumber(col) : undefined;

 405        let result = new Data({

 406            colNames:this.colNames(), 

 407            rows:this.data.map((row:any[], rowIndex:number, rows:any[][]) => { 

 408                if (c) {

 409                    row[c] = (fn)(row[c], c, rowIndex, rows); 

 410                } else {

 411                    rows[rowIndex] = (fn)(row, rowIndex, rows);

 412                }

 413                return row;

 414            }), 

 415            name:this.getName()

 416        });

 417        return result;

 444        this.meta = [];

 452     * Determines the type of data in `col`. An array of counts is created for all

 453     * encountered types, sorted by descending frequency. THe most likely type in position 0

 454     * of the array is returned.

 455     * @param col the index of the column to be typed. 

 456     * @return the most likely type of data in `col`.

 457     */

 458    private findTypes(col:ColumnReference):string {

 459        const m = this.getMeta(col);

 460        const types:TypeStruct[] = [];

 461        Object.keys(Data.type).forEach((t:string) => {

 462            const ts = { type: Data.type[t], count: 0 }; 

 463            types.push(ts);

 464            types[Data.type[t]] = ts;

 465        });

 466        for (let v of this.allRows(col)) {

 467            const t = this.findType(v);

 468            if (t !== null) { types[t].count++; }

 469        }

 470        types.sort(function(a:TypeStruct, b:TypeStruct) { 

 471            if (a.type==='currency'&&a.count>0) { return -1; }

 472            if (b.type==='currency'&&b.count>0) { return 1; }

 473            return b.count - a.count;

 474        });

 475        m.types = types;

 476        return types[0].type;

 477    }

 490    private findType(val:DataVal) {

 508            switch (strVal.toLowerCase()) {

 509                case "null": break;

 528    /**

 529     * @param val the string to convert to a date

 530     * @param limitYear the year below which the century is corrected. Defaults to 1970.

 531     * @returns a new Date object parsed from `str`.

 532     * @description returns a new Date object parsed from `str` and corrects for a difference in 

 533     * interpreting centuries between webkit and mozilla in converting strings to Dates:

 534     * The string "15/7/03" will convert to Jul 15 1903 in Mozilla and July 15 2003 in Webkit.

 535     * If `limitYear` is not specified this method uses 1970 as the decision date: 

 536     * years 00-69 will be interpreted as 2000-2069, years 70-99 as 1970-1999.

 537     * Specifying a century in the string leads to a correct parse, e.g. "15/7/1903"

 538     */

 539    private toDate(val:DataVal, limitYear=1970):Date {

 540        let d:Date;

 541        if (val instanceof Date) { d = val; }

 542        else { 

 543            d = new Date(val); 

 544            if (!isNaN(d.getTime())) {

 545                const r = new RegExp(`\\d\\d${d.getFullYear() % 100}`, 'g');

 546                if (!(val).match(r)) {

 547                    const yr = 1900 + d.getFullYear() % 100;

 548                    d.setFullYear( (yr < limitYear)? yr+100 : yr); 

 549                }

 550            }

 551        }   

 552        return d;

 553    }

 555    /**

 556     * @param type ['date' | 'percent' | 'real' | _any_] The type to cast into. In case of _any_ - i.e. `type` 

 557     * does not match any of the previous keywords, no casting occurs.

 558     * @param sample The value to cast.

 559     * @returns The result of the cast. 

 560     * @description Casts the sample to the specified data type.

 561     */

 562    private castVal(type:string, val:DataVal):DataVal {

 563        let result:DataVal = val;

 564        switch (type) {

 565            case Data.type.date:    if (val instanceof Date) { return val; }

 566                            result = this.toDate(val);

 567                            if (isNaN(result.getTime())) { result = null; }

 568                            break;

 569            case Data.type.percent: if (typeof val === 'string') {

 570                                const num = parseFloat(val);

 571                                result = (val).endsWith('%')? num/100 : num;

 572                            } else {

 573                                result = val;

 574                            }

 575                            if (isNaN(result)) { result = null; }

 576                            break;

 577            case Data.type.currency:// replace all except 'e/E', '.', '+/-' and digits

 578                            result = (typeof val === 'string')? val.replace(/[^eE\+\-\.\d]/g, '') : val;          

 579             /* falls through */

 580            case Data.type.number:  if (typeof result === 'string') { result = parseFloat(result); }

 581                            if (isNaN(result)) { result = null; }

 582                            break;

 583            default:        result = ''+val;

 584        }

 585        return result;

 586     }
